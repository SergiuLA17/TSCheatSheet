Schimbarea HTML cu ajutorul JS

                    
// codul dat genereaza in body un h4 cu textul "Text"
const htmlTag = document.createElement("h4");
htmlTag.innerHTML = "Text";
document.body.appendChild(htmlTag);
                  
                
                    
// codul dat schimba culoarea fundalului in albastru al elementului cu clasa "bg"
//variant 1
document.getElementsByClassName("bg")[0].style.backgroundColor = "blue";
//variant 2
document.querySelector('.bg').style.backgroundColor = 'blue'
                  
                
                    
// codul dat creaza elementul p si adauga classa si il adauga in body
function welcomeToMyClass(className) {

  const el = document.createElement('p')
    el.classList.add(className)
    el.innerText = className[0].toUpperCase() + className.slice(1)
    document.body.appendChild(el)

}
